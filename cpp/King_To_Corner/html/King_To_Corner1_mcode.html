1function [ move_to_corner ] = King_To_Corner( board, x, y )
 2%GENERATE_MOVES_FOR_PIECE Takes the 2d 11x11 array and returns the number
 3%of moves to each corner tl, tr, bl, br
 4%   Detailed explanation goes here
 5    
 6    move_to_corner = zeros(1,4,'int8');
 7    
 8    [r,c] = size(board);

11        return;

14%   Not sure if this kind of index checking is appropriate

17       return;

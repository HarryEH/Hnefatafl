1function [ moves_array ] = Generate_Moves_For_Piece( board,x,y)
 2
 3    moves_array = zeros(20,2,'int8');
 4    
 5    if ( x > 11 || x < 1)
 6       return; 
 7    end
 8    
 9    if ( y > 11 || y < 1)
10       return; 
11    end
12    
13    if ( board(x,y) == 0 )
14        return;
15    end
16    
17    left_t  = true;
18    up_t    = true;
19    down_t  = true;
20    right_t = true;
21    
22    move_counter = 1;
23    
24    for i = 1:10
25        
26        if left_t || right_t || up_t || down_t  
27            if left_t
28                if ((x-i) > 0) && (board(x-i,y) ~= -1)
29                    moves_array(move_counter,1) = (x-i);
30                    moves_array(move_counter,2) = y;
31                    move_counter = move_counter + 1;
32                else 
33                    left_t = false;
34                end
35            end
36
37            if right_t
38                if ((x+i) < 12) && (board(x+1,y) ~= -1)
39                    moves_array(move_counter,1) = (x+i);
40                    moves_array(move_counter,2) = y;
41                    move_counter = move_counter + 1;
42                else 
43                    right_t = false;
44                end
45            end
46
47            if up_t
48                if ((y+i) < 12) && (board(x,y+i) ~= -1)
49                    moves_array(move_counter,1) = x;
50                    moves_array(move_counter,2) = (y+i);
51                    move_counter = move_counter + 1;
52                else 
53                    up_t = false;
54                end
55            end
56
57            if down_t
58                if ((y-i) > 0) && (board(x,y-i) ~= -1)
59                    moves_array(move_counter,1) = x;
60                    moves_array(move_counter,2) = (y-i);
61                    move_counter = move_counter + 1;
62                else 
63                    down_t = false;
64                end
65            end
66        else
67            break;
68        end    
69    end 
70end
